import static org.junit.Assert.*;
import org.junit.Test;
import java.text.ParseException;
import java.util.Date;
import javax.xml.datatype.XMLGregorianCalendar;

public class FormatterTest {

    @Test
    public void testDateToXMLGregorianCal() {
        Date javaDate = new Date();
        XMLGregorianCalendar xmlCal = Formatter.dateToXMLGregorianCal(javaDate);
        assertNotNull(xmlCal);
        assertEquals(javaDate, Formatter.xmlGregorianCalToDate(xmlCal));
    }

    @Test
    public void testDateToXMLGregorianDate() {
        Date javaDate = new Date();
        XMLGregorianCalendar xmlCal = Formatter.dateToXMLGregorianDate(javaDate);
        assertNotNull(xmlCal);
        assertNull(xmlCal.getTimezone());
        assertEquals(javaDate, Formatter.xmlGregorianCalToDate(xmlCal));
    }

    @Test
    public void testParseDate() throws ParseException {
        String dateString = "2023-09-20T10:30:00Z"; // Replace with a valid date string
        XMLGregorianCalendar xmlCal = Formatter.parseDate(dateString);
        assertNotNull(xmlCal);
        assertEquals(dateString, Formatter.formatDate(xmlCal));
    }

    @Test
    public void testFormatDate() {
        // Replace with a valid XMLGregorianCalendar and its expected formatted string
        XMLGregorianCalendar xmlCal = null;
        String expectedFormattedDate = ""; // Replace with the expected formatted date
        assertEquals(expectedFormattedDate, Formatter.formatDate(xmlCal));
    }
}
