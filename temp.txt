it('should set the Accept header and send a DELETE request', () => {
    const observe = 'body';
    service.errorHtmlUsingDELETE(observe).subscribe();

    const req = httpTestingController.expectOne(`${service['basePath']}/error`);
    expect(req.request.method).toBe('DELETE');
    
    // Check if the Accept header is correctly set
    expect(req.request.headers.get('Accept')).toBe('text/html');
    
    req.flush({}); // You can provide a response here
  });
});
