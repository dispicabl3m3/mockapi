import { Component } from '@angular/core';
import { Router } from '@angular/router';

@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss']
})
export class HeaderComponent {

  /** holds the value of selected menu item */
  selectedValue: string="";

  /** list of menu items to show */
  options=[
    {value:'/access-channel', label:'Access Channel'},
    {value:'/cotiviti', label:'Cotiviti'},
    {value:'/idr', label:'Independent Dispute Resolution (IDR)'},
    {value:'/shared-accums-inbound', label:'Shared Accums - Inbound'},
    
  ];

  constructor(private router: Router){
  }

  /** clicking on a menu item to navigate to the appropriate application 
   * every time menu option changes, we would show a default message 
   * this would insruct the user to click on a tab to continue an action
  */
  navigate(value: any){
    this.router.navigate([value],{queryParams : { showDefaultMessage: true }});
  }
}
